删除30天前的过期数据

 
opengausspostgresql数据库日志脚本

由于数据库产生的临时信息，不需要保存太久的，占用数据空间比较大，需要定时清除。

[Database]#cat query_and_delete.sql
BEGIN;

-- 查询30天之前的记录数并导出到临时表
CREATE TEMP TABLE temp_count AS
SELECT COUNT(*) AS record_count
FROM fssd_uav_prod."public"."cloud_dock_real_data"
WHERE "receive"::timestamp < NOW() - INTERVAL '30 days';

-- 删除30天之前的记录
DELETE FROM fssd_uav_prod."public"."cloud_dock_real_data"
WHERE "receive"::timestamp < NOW() - INTERVAL '30 days';

COMMIT;

-- 导出查询结果到 CSV 文件
COPY (SELECT * FROM temp_count) TO '/tmp/cloud_dock_real_data_drop_5days_before.csv' WITH CSV HEADER;

-- 删除临时表
DROP TABLE temp_count;
