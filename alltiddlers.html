计算内核参数的值：

echo "kernel.sem= cat /proc/sys/kernel/sem" && cat /proc/meminfo|grep MemTotal|awk {'print $2'*1024*0.8/$(getconf PAGE_SIZE)}|awk '{printf "kernel.shmall=%d\n",$1}' && cat /proc/meminfo|grep MemTotal| awk '{printf "kernel.shmmax=%d\n",$2*1024*0.8}' && echo "kernel.shmmni =cat /proc/sys/kernel/shmmni"

编辑 /etc/sysctl.conf ，添加如下信息。shmall、shmmax直接取上一步计算的值，shmmni、sem若计算值大于下方样例，则可#注释下方这两行参数，保持原有默认值即可： fs.aio-max-nr=1048576 fs.file-max= 76724600 kernel.sem = 250 64000 250 1024 kernel.shmall = 3643361024 # pages, 80% MEM or higher kernel.shmmax = 86641802241024 # bytes, 80% MEM or higher kernel.shmmni = 8192 net.core.netdev_max_backlog = 10000 net.core.rmem_default = 262144 net.core.rmem_max = 4194304 net.core.wmem_default = 262144 net.core.wmem_max = 4194304 net.core.somaxconn = 4096 net.ipv4.tcp_fin_timeout = 5 vm.overcommit_memory = 0 vm.swappiness = 10 net.ipv4.ip_local_port_range = 40000 65535 fs.nr_open = 20480000 kernel.core_pattern = /data/core_pattern/core-%e-%p-%t

执行以下命令，使配置生效（注意若部分参数报错，可酌情修改参数配置）： sysctl -p

#!/bin/bash

# 检查操作系统类型
if [ -f /etc/os-release ]; then
    . /etc/os-release
    OS=$ID
else
    echo "无法识别操作系统类型。"
    exit 1
fi

# 安装依赖
if [[ "$OS" == "ubuntu" || "$OS" == "debian" ]]; then
    # Debian/Ubuntu 系列
    sudo apt update
    if ! dpkg -l | grep -q build-essential; then
        sudo apt install -y build-essential wget
    fi
elif [[ "$OS" == "centos" || "$OS" == "openEuler" ]]; then
    # CentOS/openEuler 系列
    sudo yum install -y epel-release
    if ! rpm -qa | grep -q "pcre"; then
        sudo yum install -y pcre pcre-devel #gcc make zlib-devel openssl-devel libxml2-devel wget
    fi
else
    echo "不支持的操作系统类型：$OS"
    exit 1
fi

# 定义要检查的文件和目录
cd /usr/local/src
dir="zabbix-7.0.2"

# 检查目录是否存在,不存在则下载
if [ -d "$dir" ]; then
    echo "Directory '$dir' exists."
else
    wget http://ibenyi.tk/zabbix-7.0.2.tar.gz
    tar zxvf zabbix-7.0.2.tar.gz
fi

# 安装 Zabbix
cd zabbix-7.0.2
./configure --enable-agent
if [ $? -ne 0 ]; then
    echo "Configure 失败，脚本终止。"
    exit 1
fi

make
if [ $? -ne 0 ]; then
    echo "Make 失败，脚本终止。"
    exit 1
fi

sudo make install
if [ $? -ne 0 ]; then
    echo "Make install 失败，脚本终止。"
    exit 1
fi

# 创建 Zabbix 用户和组
if [[ "$OS" == "ubuntu" || "$OS" == "debian" ]]; then
    addgroup --system --quiet zabbix
    adduser --quiet --system --disabled-login --ingroup zabbix --home /var/lib/zabbix --no-create-home zabbix
else
    groupadd --system zabbix
    useradd --system -g zabbix -d /usr/lib/zabbix -s /sbin/nologin -c "Zabbix Monitoring System" zabbix
fi

# 创建日志和运行目录
sudo mkdir -p /var/log/zabbix
sudo mkdir -p /var/run/zabbix
sudo chown zabbix:zabbix /usr/local/sbin/zabbix_agentd
sudo chown -R zabbix:zabbix /usr/local/etc/zabbix_agentd.conf /usr/local/etc/zabbix_agentd.conf.d /var/log/zabbix /var/run/zabbix
ip=$(hostname -I |awk {'print $1'})

# 修改配置文件
sudo sed -i '12i PidFile=/tmp/zabbix_agentd.pid' /usr/local/etc/zabbix_agentd.conf
sudo sed -i 's/^Server=127.0.0.1/Server=192.168.1.222/' /usr/local/etc/zabbix_agentd.conf
sudo sed -i 's#^LogFile=/tmp/zabbix_agentd.log#LogFile=/var/log/zabbix/zabbix_agentd.log#' /usr/local/etc/zabbix_agentd.conf
sudo sed -i 's/^# DebugLevel=3/DebugLevel=3/' /usr/local/etc/zabbix_agentd.conf
sudo sed -i 's/^# AllowRoot=0/AllowRoot=1/' /usr/local/etc/zabbix_agentd.conf
sudo sed -i 's/^# HostnameItem=system.hostname/HostnameItem=system.hostname/' /usr/local/etc/zabbix_agentd.conf
sudo sed -i "s/^Hostname=Zabbix server/Hostname=$ip/" /usr/local/etc/zabbix_agentd.conf
sudo sed -i 's/^# User=zabbix/User=zabbix/' /usr/local/etc/zabbix_agentd.conf
sudo sed -i 's/^# Include=\/usr\/local\/etc\/zabbix_agentd.conf.d\/\*.conf/Include=\/usr\/local\/etc\/zabbix_agentd.conf.d\/\*.conf/' /usr/local/etc/zabbix_agentd.conf
sudo sed -i 's/^# UnsafeUserParameters=0/UnsafeUserParameters=1/' /usr/local/etc/zabbix_agentd.conf
sudo sed -i 's/^# UserParameterDir=/UserParameterDir=\/usr\/local\/etc\/zabbix\/zabbix_agentd.conf.d/' /usr/local/etc/zabbix_agentd.conf
sudo sed -i 's/^ServerActive=127.0.0.1/ServerActive=192.168.1.222/' /usr/local/etc/zabbix_agentd.conf
sudo echo 'AllowKey=system.run[*]' >> /usr/local/etc/zabbix_agentd.conf

# 设置禁用selinux
sestatus
if [[ "$(getenforce)" != "Disabled" && "$(getenforce)" != "disabled" ]]; then
    sudo sed -i 's/^SELINUX=.*/SELINUX=disabled/' /etc/selinux/config
    sudo setenforce 0
fi

if [[ "$OS" == "ubuntu" || "$OS" == "debian" ]]; then
    # Debian/Ubuntu 系列开放防火墙
    ufw_status=$(sudo ufw status | grep -o "Status: active")
    echo "UFW status: $ufw_status"

    # 如果 ufw 在运行，则开放 10050/tcp 端口
    if [ "$ufw_status" = "Status: active" ]; then
        echo "Opening port 10050/tcp..."
        sudo ufw allow 10050/tcp
        echo "Port 10050/tcp has been opened."

        # 验证端口是否开放
        sudo ufw status | grep 10050/tcp
        if [ $? -eq 0 ]; then
            echo "Port 10050/tcp is successfully opened."
        else
            echo "Failed to open port 10050/tcp."
        fi
    else
        echo "UFW is not running."
    fi
elif [[ "$OS" == "centos" || "$OS" == "openEuler" ]]; then
    # CentOS/Redhat/Openeuler开放防火墙方式

    # 检查 firewalld 状态
    firewalld_status=$(sudo systemctl is-active firewalld)
    echo "Firewalld status: $firewalld_status"

    # 如果 firewalld 在运行，则开放 10050/tcp 端口
    if [ "$firewalld_status" = "active" ]; then
        echo "Opening port 10050/tcp..."
        sudo firewall-cmd --zone=public --add-port=10050/tcp --permanent
        sudo firewall-cmd --reload

CREATE TEMP TABLE temp_count AS
SELECT COUNT(*) AS record_count
FROM fssd_uav_prod."public"."cloud_dock_real_data"
WHERE "receive"::timestamp < NOW() - INTERVAL '30 days';

# 创建备份目录

# 执行清除cloud_dock_real_data表5天前的数据
psql -h $HOST -p $PORT -U $USER -d fssd_uav_prod -f query_and_delete.sql -W "pasword"

# 移动 CSV 文件到备份目录
mv /tmp/cloud_dock_real_data_drop_5days_before.csv "$BACKUP_DIR/cloud_dock_real_data_drop_5days_before.log"

# 清理临时文件
rm -f /tmp/cloud_dock_real_data_drop_5days_before.csv

# 获取数据库列表
DATABASES=$(psql -h $HOST -p $PORT -U $USER -d vastbase -t -c "SELECT datname FROM pg_database WHERE datname LIKE '%prod%' AND datistemplate = false;" -W "pasword")

# 循环备份每个数据库
for DB in $DATABASES; do
  DB=$(echo $DB | xargs) # 去除空格
  echo "正在导出数据库：$DB"

  # 执行备份
  pg_dump -h $HOST -p $PORT -U $USER -F c -b -v -W "pasword" -f "${BACKUP_DIR}/${DB}.${DATE}.backup" $DB

  # 检查备份是否成功
  if [ $? -eq 0 ]; then
    echo "数据库 $DB 导出成功" >> "${BACKUP_DIR}/backup.log"
  else
    echo "数据库 $DB 导出失败" >> "${BACKUP_DIR}/backup.log"
  fi
done


#!/bin/bash

# 数据库连接信息
DB_USER="postgreq"
DB_HOST="127.0.0.1"
DB_PORT="5432"
BACKUP_DIR="/backup_dir"

# 提示用户输入需要还原数据库的日期
read -p "请输入需要还原数据库的日期 (格式: YYYY-MM-DD): " datetime

# 检查备份目录是否存在
BACKUP_PATH="$BACKUP_DIR/$datetime"
if [ ! -d "$BACKUP_PATH" ]; then
    echo "没有当前日期的备份文件。"
    exit 1
fi

# 询问是否要还原目录下的所有数据库
read -p "是否要还原目录下的所有数据库？(y/n): " restore_all

if [[ "$restore_all" =~ ^[Yy]$ ]]; then
    # 还原所有数据库
    for backup_file in "$BACKUP_PATH"/*.backup; do
        # 提取文件名
        filename="${backup_file##*/}"
        # 提取数据库名
        db_name="${filename%.*}"
        db_name="${db_name%.*}"
        echo "还原数据库 $db_name..."
        pg_restore -h "$DB_HOST" -p "$DB_PORT" -U "$DB_USER" -W "password" -d "$db_name" "$backup_file"
        if [ $? -ne 0 ]; then
            echo "还原数据库 $db_name 失败。"
            exit 1
        fi
    done
else
    # 提示输入需要还原的数据库
    read -p "请输入需要还原的数据库名称: " db_name
    backup_file="$BACKUP_PATH/$db_name.$datetime.backup"
    if [ ! -f "$backup_file" ]; then
        echo "没有找到数据库 $db_name 的备份文件。"
        exit 1
    fi
    echo "还原数据库 $db_name..."
    pg_restore -h "$DB_HOST" -p "$DB_PORT" -U "$DB_USER" -W "password" -d "$db_name" "$backup_file"
    if [ $? -ne 0 ]; then
        echo "还原数据库 $db_name 失败。"
        exit 1
    fi
fi

echo "数据库还原完成。"
